### 1.4 关键适配点

| 适配点 | 实现方式 | 说明 |
|--------|---------|------|
| **触摸区域** | 最小44px高度 | iOS Human Interface Guidelines |
| **字体大小** | 输入框16px | 防止iOS自动缩放 |
| **虚拟键盘** | 使用vh单位 | 避免绝对定位 |
| **卡片阴影** | 移动端去除 | 提升性能 |

## 4. 数据列表页面适配

### 4.1 通用适配方案

**问题**：Ant Design Table在移动端显示不佳

**解决方案**：响应式表格组件

```tsx
/**
 * 响应式表格组件
 * 移动端：Card列表
 * PC端：Table表格
 */
interface ResponsiveTableProps<T> {
  data: T[];
  columns: ColumnsType<T>;
  cardRender: (item: T) => React.ReactNode;
  loading?: boolean;
  pagination?: false | PaginationConfig;
}

function ResponsiveTable<T extends { id: string | number }>({
  data,
  columns,
  cardRender,
  loading,
  pagination,
}: ResponsiveTableProps<T>) {
  const { isMobile } = useDevice();
  
  // 移动端：Card列表
  if (isMobile) {
    return (
      <div className="responsive-table-mobile">
        {loading ? (
          <Spin />
        ) : (
          <>
            {data.map(item => (
              <div key={item.id} className="data-card">
                {cardRender(item)}
              </div>
            ))}
            
            {/* 移动端分页 */}
            {pagination && (
              <div className="mobile-pagination">
                <Button
                  disabled={pagination.current === 1}
                  onClick={() => pagination.onChange?.(pagination.current! - 1)}
                >
                  上一页
                </Button>
                <span>
                  {pagination.current} / {Math.ceil(pagination.total! / pagination.pageSize!)}
                </span>
                <Button
                  disabled={pagination.current === Math.ceil(pagination.total! / pagination.pageSize!)}
                  onClick={() => pagination.onChange?.(pagination.current! + 1)}
                >
                  下一页
                </Button>
              </div>
            )}
          </>
        )}
      </div>
    );
  }
  
  // PC端：Table
  return (
    <Table<T>
      dataSource={data}
      columns={columns}
      loading={loading}
      pagination={pagination}
    />
  );
}
```

const ResponsiveChart: React.FC<{ options: EChartOption }> = ({ options }) => {
  const chartRef = useRef<HTMLDivElement>(null);
  const { isMobile } = useDevice();
  
  useEffect(() => {
    if (!chartRef.current) return;
    
    const chart = echarts.init(chartRef.current);
    
    // 移动端优化
    const mobileOptions = isMobile
      ? {
          ...options,
          grid: {
            left: '10%',
            right: '10%',
            bottom: '15%',
          },
          xAxis: {
            ...options.xAxis,
            axisLabel: {
              rotate: 45,
              fontSize: 10,
            },
          },
          yAxis: {
            ...options.yAxis,
            axisLabel: {
              fontSize: 10,
            },
          },
          tooltip: {
            trigger: 'axis',
            axisPointer: {
              type: 'cross',
            },
          },
        }
      : options;
    
    chart.setOption(mobileOptions);
    
    // 窗口resize时重绘
    const handleResize = () => chart.resize();
    window.addEventListener('resize', handleResize);
    
    return () => {
      window.removeEventListener('resize', handleResize);
      chart.dispose();
    };
  }, [options, isMobile]);
  
  return (
    <div
      ref={chartRef}
      style={{ height: isMobile ? '250px' : '400px' }}
    />
  );
};

/**

const useChartTouchOptimization = (chartRef: RefObject<HTMLDivElement>) => {
  useEffect(() => {
    const chartDom = chartRef.current;
    if (!chartDom) return;
    
    let startX = 0;
    let startY = 0;
    
    const handleTouchStart = (e: TouchEvent) => {
      startX = e.touches[0].clientX;
      startY = e.touches[0].clientY;
    };
    
    const handleTouchEnd = (e: TouchEvent) => {
      const endX = e.changedTouches[0].clientX;
      const endY = e.changedTouches[0].clientY;
      
      const diffX = endX - startX;
      const diffY = endY - startY;
      
      // 横向滑动切换时间范围
      if (Math.abs(diffX) > Math.abs(diffY) && Math.abs(diffX) > 50) {
        if (diffX > 0) {
          // 右滑：切换到上一个时间范围
          console.log('切换到上一个时间范围');
        } else {
          // 左滑：切换到下一个时间范围
          console.log('切换到下一个时间范围');
        }
      }
    };
    
    chartDom.addEventListener('touchstart', handleTouchStart, { passive: true });
    chartDom.addEventListener('touchend', handleTouchEnd, { passive: true });
    
    return () => {
      chartDom.removeEventListener('touchstart', handleTouchStart);
      chartDom.removeEventListener('touchend', handleTouchEnd);
    };
  }, [chartRef]);
};
