## 1. 性能优化策略

### 1.1 优化原则

```
┌─────────────────────────────────────────────────────────────┐
│                    性能优化原则                               │
├─────────────────────────────────────────────────────────────┤
│  1. 测量优先，优化在后                                        │
│     - 使用Lighthouse、Chrome DevTools测量性能               │
│     - 找到性能瓶颈再优化                                     │
│                                                           │
│  2. 渐进增强，优雅降级                                        │
│     - 先保证功能正常，再优化性能                             │
│     - 不支持的浏览器优雅降级                                 │
│                                                           │
│  3. 移动端优先                                                │
│     - 首先考虑移动端性能                                     │
│     - 再考虑PC端增强                                        │
│                                                           │
│  4. 真实设备测试                                              │
│     - 模拟器结果仅供参考                                     │
│     - 必须在真实设备上验证                                    │
└─────────────────────────────────────────────────────────────┘
```

### 1.2 关键性能指标

| 指标 | 说明 | 优秀 | 需改进 |
|------|------|------|--------|
| **FCP** | 首次内容绘制 | < 1.8s | > 3.0s |
| **LCP** | 最大内容绘制 | < 2.5s | > 4.0s |
| **TBT** | 总阻塞时间 | < 200ms | > 600ms |
| **CLS** | 累积布局偏移 | < 0.1 | > 0.25 |
| **FID** | 首次输入延迟 | < 100ms | > 300ms |
| **TTI** | 可交互时间 | < 3.8s | > 7.3s |

## 2. 代码分割优化

### 2.1 路由级代码分割

**原理**：每个路由单独打包，按需加载

```tsx
/**
 * 路由级代码分割
 * 使用React.lazy和Suspense
 */

import { lazy, Suspense } from 'react';
import { Routes, Route } from 'react-router-dom';
import LoadingSpinner from '@/components/LoadingSpinner';

// 懒加载页面组件
const DeveloperDashboard = lazy(() => import('@/pages/developer/Dashboard'));
const DeveloperKeys = lazy(() => import('@/pages/developer/Keys'));
const DeveloperQuota = lazy(() => import('@/pages/developer/Quota'));
const AdminDashboard = lazy(() => import('@/pages/admin/Dashboard'));
const AdminUsers = lazy(() => import('@/pages/admin/Users'));

/**
 * 带加载指示器的懒加载组件
 */
const LazyLoad: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  return (
    <Suspense 
      fallback={
        <div className="loading-container">
          <LoadingSpinner />
          <p>加载中...</p>
        </div>
      }
    >
      {children}
    </Suspense>
  );
};

/**
 * 路由配置
 */
const AppRoutes: React.FC = () => {
  return (
    <Routes>
      {/* 开发者路由 */}
      <Route path="/" element={<LazyLoad><DeveloperDashboard /></LazyLoad>} />
      <Route path="/developer/keys" element={<LazyLoad><DeveloperKeys /></LazyLoad>} />
      <Route path="/developer/quota" element={<LazyLoad><DeveloperQuota /></LazyLoad>} />
      
      {/* 管理员路由 */}
      <Route path="/admin" element={<LazyLoad><AdminDashboard /></LazyLoad>} />
      <Route path="/admin/users" element={<LazyLoad><AdminUsers /></LazyLoad>} />
    </Routes>
  );
};
```

## 3. 懒加载优化

### 3.1 图片懒加载

```tsx
/**
 * 图片懒加载组件
 * 使用Intersection Observer API
 */

import { useRef, useEffect, useState } from 'react';

interface LazyImageProps {
  src: string;
  alt: string;
  placeholder?: string;
  className?: string;
}

const LazyImage: React.FC<LazyImageProps> = ({
  src,
  alt,
  placeholder = 'data:image/gif;base64,R0lGODlhAQABAIAAAAAAAP///yH5BAEAAAAALAAAAAABAAEAAAIBRAA7',
  className,
}) => {
  const [loaded, setLoaded] = useState(false);
  const [inView, setInView] = useState(false);
  const imgRef = useRef<HTMLImageElement>(null);
  
  useEffect(() => {
    const observer = new IntersectionObserver(
      (entries) => {
        entries.forEach((entry) => {
          if (entry.isIntersecting) {
            setInView(true);
            observer.unobserve(entry.target);
          }
        });
      },
      { rootMargin: '50px' } // 提前50px开始加载
    );
    
    if (imgRef.current) {
      observer.observe(imgRef.current);
    }
    
    return () => observer.disconnect();
  }, []);
  
  return (
    <img
      ref={imgRef}
      src={inView ? src : placeholder}
      alt={alt}
      className={`${className} ${loaded ? 'loaded' : 'loading'}`}
      onLoad={() => setLoaded(true)}
    />
  );
};

/**
 * 使用示例
 */
const ImageGallery: React.FC<{ images: string[] }> = ({ images }) => {
  return (
    <div className="image-gallery">
      {images.map((url, index) => (
        <LazyImage
          key={index}
          src={url}
          alt={`Image ${index + 1}`}
          className="gallery-image"
        />
      ))}
    </div>
  );
};
```

### 3.2 虚拟列表（长列表优化）

```tsx
/**
 * 虚拟列表组件
 * 使用react-window
 */

import { FixedSizeList as List } from 'react-window';
import { useDevice } from '@/hooks/useDevice';

interface VirtualListProps<T> {
  data: T[];
  renderItem: (item: T, index: number) => React.ReactNode;
  itemHeight?: number;
  containerHeight?: number;
}

function VirtualList<T>({
  data,
  renderItem,
  itemHeight = 60,
  containerHeight = 600,
}: VirtualListProps<T>) {
  const { isMobile } = useDevice();
  
  // 移动端使用较小的高度
  const actualHeight = isMobile ? Math.min(containerHeight, 500) : containerHeight;
  const actualItemHeight = isMobile ? Math.min(itemHeight, 80) : itemHeight;
  
  // 如果数据量小，不使用虚拟列表
  if (data.length < 50) {
    return (
      <div className="simple-list">
        {data.map((item, index) => (
          <div key={index}>{renderItem(item, index)}</div>
        ))}
      </div>
    );
  }
  
  // 数据量大，使用虚拟列表
  const Row = ({ index, style }: { index: number; style: React.CSSProperties }) => (
    <div style={style}>
      {renderItem(data[index], index)}
    </div>
  );
  
  return (
    <List
      height={actualHeight}
      itemCount={data.length}
      itemSize={actualItemHeight}
      width="100%"
      overscanCount={5} // 预渲染5个
    >
      {Row}
    </List>
  );
}

/**
 * 使用示例
 */
const ApiKeyList: React.FC<{ keys: ApiKey[] }> = ({ keys }) => {
  const renderKey = (key: ApiKey, index: number) => (
    <div className="key-item">
      <span>{key.name}</span>
      <Tag color={key.status === 'active' ? 'green' : 'red'}>
        {key.status}
      </Tag>
    </div>
  );
  
  return <VirtualList data={keys} renderItem={renderKey} />;
};
```

## 5. 网络优化

### 5.1 API请求优化

```typescript
/**
 * API请求去重
 */

class RequestDeduplicator {
  private pendingRequests: Map<string, Promise<any>> = new Map();
  
  async fetch(url: string, options?: RequestInit): Promise<any> {
    const key = `${url}-${JSON.stringify(options)}`;
    
    // 如果已有相同的请求在进行，返回同一个Promise
    if (this.pendingRequests.has(key)) {
      return this.pendingRequests.get(key);
    }
    
    // 发起新请求
    const promise = fetch(url, options).then(res => res.json());
    this.pendingRequests.set(key, promise);
    
    // 请求完成后删除
    promise.finally(() => {
      this.pendingRequests.delete(key);
    });
    
    return promise;
  }
}

const deduplicator = new RequestDeduplicator();

/**
 * 请求缓存
 */

class ApiCache {
  private cache: Map<string, { data: any; expiry: number }> = new Map();
  
  async get(url: string, ttl: number = 60000): Promise<any> {
    const cached = this.cache.get(url);
    
    // 如果缓存未过期，返回缓存数据
    if (cached && Date.now() < cached.expiry) {
      return cached.data;
    }
    
    // 缓存过期或不存在，发起请求
    const data = await fetch(url).then(res => res.json());
    this.cache.set(url, { data, expiry: Date.now() + ttl });
    
    return data;
  }
  
  invalidate(url: string) {
    this.cache.delete(url);
  }
}

const apiCache = new ApiCache();

/**
 * 使用示例
 */
const fetchUserData = async (userId: string) => {
  // 缓存1分钟
  return apiCache.get(`/api/users/${userId}`, 60000);
};
```

### 5.3 使用Service Worker缓存

```typescript
/**
 * Service Worker注册
 */

// src/serviceWorkerRegistration.ts
export function register() {
  if ('serviceWorker' in navigator) {
    window.addEventListener('load', () => {
      const swUrl = `${process.env.PUBLIC_URL}/service-worker.js`;
      
      navigator.serviceWorker
        .register(swUrl)
        .then((registration) => {
          console.log('SW registered: ', registration);
        })
        .catch((error) => {
          console.error('SW registration failed: ', error);
        });
    });
  }
}

/**
 * Service Worker脚本
 */

// public/service-worker.js
const CACHE_NAME = 'api-platform-v1';
const urlsToCache = [
  '/',
  '/index.html',
  '/static/js/main.js',
  '/static/css/main.css',
];

// 安装时缓存资源
self.addEventListener('install', (event) => {
  event.waitUntil(
    caches.open(CACHE_NAME).then((cache) => {
      return cache.addAll(urlsToCache);
    })
  );
});

// 拦截请求，返回缓存
self.addEventListener('fetch', (event) => {
  event.respondWith(
    caches.match(event.request).then((response) => {
      // 如果缓存中有，返回缓存
      if (response) {
        return response;
      }
      
      // 否则发起网络请求
      return fetch(event.request).then((response) => {
        // 克隆响应
        const responseToCache = response.clone();
        
        caches.open(CACHE_NAME).then((cache) => {
          cache.put(event.request, responseToCache);
        });
        
        return response;
      });
    })
  );
});
```

```typescript
      tcp: perfData.connectEnd - perfData.connectStart,
      
      // 首字节时间（TTFB）
      ttfb: perfData.responseStart - perfData.requestStart,
      
      // DOM解析时间
      domParse: perfData.domInteractive - perfData.domLoading,
      
      // 页面完全加载时间
      load: perfData.loadEventEnd - perfData.navigationStart,
    };
    
    console.log('性能指标:', metrics);
    
    // 发送到监控服务器
    sendMetricsToServer(metrics);
  });
};

/**
 * 使用Performance Observer API
 */

const observePerformance = () => {
  // 观察长任务
  const observer = new PerformanceObserver((list) => {
    list.getEntries().forEach((entry) => {
      console.log('长任务:', entry.duration);
    });
  });
  
  observer.observe({ entryTypes: ['longtask'] });
};
```

```tsx
        domNodes: document.getElementsByTagName('*').length,
        
        // 事件监听器数量（近似）
        eventListeners: getEventListenersCount(),
      });
    };
    
    const interval = setInterval(updateMetrics, 2000);
    return () => clearInterval(interval);
  }, []);
  
  if (process.env.NODE_ENV !== 'development') {
    return null;
  }
  
  return (
    <div className="performance-monitor">
      <h4>性能监控</h4>
      <div>内存使用: {metrics.memoryUsed}MB / {metrics.memoryTotal}MB</div>
      <div>DOM节点: {metrics.domNodes}</div>
```
