## 第二章 快速部署

### 2.1 一键部署脚本

```bash
#!/bin/bash
# deploy.sh - 一键部署脚本

set -e

# 配置
APP_NAME="api-service"
REGISTRY="registry.example.com"
VERSION="${1:-latest}"
DATA_DIR="/data/${APP_NAME}"
ENV_FILE="${DATA_DIR}/.env"

# 颜色定义
RED='\033[0;31m'
GREEN='\033[0;32m'
NC='\033[0m'

log_info() {
    echo -e "${GREEN}[INFO]${NC} $1"
}

log_error() {
    echo -e "${RED}[ERROR]${NC} $1"
}

# 检查Docker
check_docker() {
    if ! command -v docker &> /dev/null; then
        log_error "Docker未安装，请先安装Docker"
        exit 1
    fi
    log_info "Docker检查通过"
}

# 创建数据目录
create_dirs() {
    log_info "创建数据目录..."
    mkdir -p ${DATA_DIR}/{data,logs,backup,ssl}
    chmod -R 755 ${DATA_DIR}
}

# 配置环境变量
setup_env() {
    log_info "配置环境变量..."
    if [ ! -f ${ENV_FILE} ]; then
        cp .env.example ${ENV_FILE}
        log_info "请编辑 ${ENV_FILE} 配置数据库密码等敏感信息"
    fi
}

# 拉取镜像
pull_images() {
    log_info "拉取Docker镜像..."
    docker-compose pull
}

# 启动服务
start_services() {
    log_info "启动服务..."
    docker-compose up -d
    
    # 等待服务就绪
    log_info "等待服务启动..."
    sleep 10
    
    # 检查服务状态
    if docker-compose ps | grep -q "Up"; then
        log_info "服务启动成功!"
    else
        log_error "服务启动失败，请检查日志"
        docker-compose logs
        exit 1
    fi
}

# 主流程
main() {
    log_info "开始部署 ${APP_NAME} v${VERSION}..."
    
    check_docker
    create_dirs
    setup_env
    pull_images
    start_services
    
    log_info "部署完成!"
    log_info "API地址: http://localhost:8000"
    log_info "Swagger文档: http://localhost:8000/docs"
}

main "$@"
```

### 3.2 Docker Compose配置

**`docker-compose.yml`**：

```yaml
version: '3.8'

services:
  # API服务
  api:
    build:
      context: .
      dockerfile: Dockerfile.api
    container_name: api-service
    restart: always
    ports:
      - "8000:8000"
    environment:
      - APP_ENV=production
    env_file:
      - ${DATA_DIR}/.env
    volumes:
      - ./data:/app/data
      - ./logs:/app/logs
    depends_on:
      - postgres
      - redis
      - minio
    networks:
      - api-network
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8000/health"]
      interval: 30s
      timeout: 10s
      retries: 3

  # Nginx反向代理
  nginx:
    image: nginx:1.25-alpine
    container_name: api-nginx
    restart: always
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./config/nginx/nginx.conf:/etc/nginx/nginx.conf:ro
      - ./config/nginx/conf.d:/etc/nginx/conf.d:ro
      - ${DATA_DIR}/ssl:/etc/nginx/ssl:ro
    depends_on:
      - api
    networks:
      - api-network

  # PostgreSQL数据库
  postgres:
    image: postgres:14-alpine
    container_name: api-postgres
    restart: always
    environment:
      POSTGRES_DB: api_service
      POSTGRES_USER: api_user
      POSTGRES_PASSWORD: ${DB_PASSWORD}
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./scripts/init_db.sql:/docker-entrypoint-initdb.d/init.sql:ro
    networks:
      - api-network
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U api_user -d api_service"]
      interval: 30s
      timeout: 10s
      retries: 3

  # Redis缓存
  redis:
    image: redis:7-alpine
    container_name: api-redis
    restart: always
    command: redis-server --requirepass ${REDIS_PASSWORD}
    volumes:
      - redis_data:/data
    networks:
      - api-network
    healthcheck:
      test: ["CMD", "redis-cli", "-a", "${REDIS_PASSWORD}", "ping"]
      interval: 30s
      timeout: 10s
      retries: 3

  # MinIO对象存储
  minio:
    image: minio/minio:latest
    container_name: api-minio
    restart: always
    command: server /data --console-address ":9001"
    environment:
      MINIO_ROOT_USER: ${MINIO_ACCESS_KEY}
      MINIO_ROOT_PASSWORD: ${MINIO_SECRET_KEY}
    volumes:
      - minio_data:/data
    networks:
      - api-network
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:9000/minio/health/live"]
      interval: 30s
      timeout: 10s
      retries: 3

volumes:
  postgres_data:
  redis_data:
  minio_data:

networks:
  api-network:
    driver: bridge
```

```bash
# 查看连接数
docker exec api-postgres psql -U api_user -d api_service -c "SELECT count(*) FROM pg_stat_activity;"
```

#### 4.3.1 数据库迁移

> **重要**：每次版本更新后，都需要检查并执行数据库迁移脚本。

```bash
# 进入项目目录
cd /path/to/api-platform

# 安装依赖（如果需要）
pip install -r requirements.txt

# 执行数据库迁移
python migrations/add_user_fields.py          # 用户字段扩展
python migrations/add_encrypted_key.py         # API Key 加密字段

# 如果需要重新初始化测试数据（仅开发/测试环境）
python scripts/init_db_with_data.py --drop
```

**迁移脚本说明**：

| 脚本 | 功能 | 影响 |
|------|------|------|
| `add_user_fields.py` | 添加 username、role、permissions 字段 | 新增字段 |
| `add_encrypted_key.py` | 添加 encrypted_key 字段，支持查看 API Key | 新增字段 |

**注意事项**：
- 生产环境执行迁移前，请先备份数据库
- 旧数据可能不支持某些新功能（如旧 API Key 无法查看明文）
- 如遇迁移失败，请查看错误日志并联系技术支持

### 5.3 告警脚本

```bash
#!/bin/bash
# alert.sh - 监控告警脚本

WEBHOOK_URL="https://oapi.dingtalk.com/robot/send?access_token=xxx"
ALERT_EMAIL="admin@example.com"

send_alert() {
    local level=$1
    local message=$2
    
    # 发送钉钉通知
    curl -X POST "${WEBHOOK_URL}" \
        -H "Content-Type: application/json" \
        -d "{\"msgtype\":\"text\",\"text\":{\"content\":\"[${level}] API服务告警: ${message}\"}}"
    
    # 发送邮件
    echo "${message}" | mail -s "[${level}] API服务告警" ${ALERT_EMAIL}
}

# 检查API健康
check_api() {
    response=$(curl -s -o /dev/null -w "%{http_code}" http://localhost:8000/health)
    if [ $response -ne 200 ]; then
        send_alert "CRITICAL" "API服务不可用，HTTP状态码: ${response}"
    fi
}

# 检查磁盘空间
check_disk() {
    usage=$(df / | tail -1 | awk '{print $5}' | sed 's/%//')
    if [ $usage -gt 80 ]; then
        send_alert "WARNING" "磁盘使用率: ${usage}%"
    fi
}

# 检查内存
check_memory() {
    usage=$(free | grep Mem | awk '{printf("%.0f"), $3/$2 * 100}')
    if [ $usage -gt 85 ]; then
        send_alert "WARNING" "内存使用率: ${usage}%"
    fi
}

# 主流程
check_api
check_disk
check_memory
```

## 第六章 故障处理

### 6.1 常见故障及处理

| 故障现象 | 可能原因 | 解决方案 |
|----------|----------|----------|
| 服务无法启动 | 端口被占用 | 检查并释放端口 |
| 数据库连接失败 | 数据库未就绪 | 等待数据库启动或检查配置 |
| API返回500 | 代码错误 | 查看日志，定位问题 |
| 上传文件失败 | MinIO配置错误 | 检查MinIO配置和桶权限 |
| SSL证书错误 | 证书过期或配置错误 | 更新证书或检查配置 |

### 6.2 故障排查流程

```
1. 查看服务状态
   docker-compose ps

2. 查看日志
   docker-compose logs -f --tail=200

3. 检查配置
   检查 .env 和 nginx.conf

4. 检查资源
   docker stats

5. 重启服务
   docker-compose restart

6. 如仍无法解决
   查看完整日志，联系技术支持
```

### 6.3 紧急回滚

```bash
#!/bin/bash
# rollback.sh - 紧急回滚脚本

set -e

BACKUP_DIR="/data/backup"
CURRENT_TAG="${1}"

if [ -z "$CURRENT_TAG" ]; then
    echo "Usage: $0 <tag>"
    exit 1
fi

log_info() {
    echo "[INFO] $1"
}

log_error() {
    echo "[ERROR] $1"
}

# 回滚镜像
rollback_images() {
    log_info "从备份拉取镜像..."
    docker pull ${REGISTRY}/api:${CURRENT_TAG}
}

# 停止当前服务
stop_services() {
    log_info "停止当前服务..."
    docker-compose down
}

# 恢复数据库
restore_database() {
    if [ -f "${BACKUP_DIR}/db_${CURRENT_TAG}.sql" ]; then
        log_info "恢复数据库..."
        docker-compose up -d postgres
        sleep 10
        cat "${BACKUP_DIR}/db_${CURRENT_TAG}.sql" | docker exec -i api-postgres psql -U api_user api_service
    fi
}

# 启动服务
start_services() {
    log_info "启动服务..."
    CURRENT_TAG=${CURRENT_TAG} docker-compose up -d
}

# 主流程
main() {
    log_info "开始回滚到版本 ${CURRENT_TAG}..."
    stop_services
    rollback_images
    restore_database
    start_services
    log_info "回滚完成!"
}

main "$@"
```

## 第七章 备份恢复

### 7.1 自动备份脚本

```bash
#!/bin/bash
# backup.sh - 自动备份脚本

set -e

BACKUP_DIR="/data/backup"
DATE=$(date +%Y%m%d_%H%M%S)
KEEP_DAYS=30

# 创建备份目录
mkdir -p ${BACKUP_DIR}/{database,files,configs}

# 备份数据库
backup_database() {
    echo "备份数据库..."
    docker exec api-postgres pg_dump -U api_user api_service | gzip > ${BACKUP_DIR}/database/db_${DATE}.sql.gz
}

# 备份上传文件
backup_files() {
    echo "备份上传文件..."
    tar -czf ${BACKUP_DIR}/files/files_${DATE}.tar.gz /data/api-service/uploads/
}

# 备份配置
backup_configs() {
    echo "备份配置文件..."
    tar -czf ${BACKUP_DIR}/configs/configs_${DATE}.tar.gz /data/api-service/config/
}

# 清理旧备份
cleanup_old() {
    echo "清理旧备份..."
    find ${BACKUP_DIR} -name "*.gz" -mtime +${KEEP_DAYS} -delete
}

# 主流程
backup_database
backup_files
backup_configs
cleanup_old

echo "备份完成: ${DATE}"
```

### 7.2 恢复操作

```bash
#!/bin/bash
# restore.sh - 数据恢复脚本

set -e

BACKUP_DIR="/data/backup"
DATE="${1}"

if [ -z "$DATE" ]; then
    echo "Usage: $0 <backup_date>"
    exit 1
fi

# 停止服务
docker-compose down

# 恢复数据库
restore_database() {
    echo "恢复数据库..."
    gunzip -c ${BACKUP_DIR}/database/db_${DATE}.sql.gz | docker exec -i api-postgres psql -U api_user api_service
}

# 恢复文件
restore_files() {
    echo "恢复文件..."
    tar -xzf ${BACKUP_DIR}/files/files_${DATE}.tar.gz -C /
}

# 启动服务
docker-compose up -d

echo "恢复完成: ${DATE}"
```
