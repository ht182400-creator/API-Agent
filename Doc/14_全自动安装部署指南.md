## 二.1 数据库迁移（重要）

> **每次版本更新后都需要执行数据库迁移脚本**

```bash
# 进入项目目录
cd /opt/api-platform

# 方式一：使用 Docker（推荐）
docker exec api-platform-python python migrations/add_user_fields.py
docker exec api-platform-python python migrations/add_encrypted_key.py

# 方式二：直接运行（需要安装依赖）
pip install -r requirements.txt
python migrations/add_user_fields.py
python migrations/add_encrypted_key.py
```

**迁移脚本说明**：

| 脚本 | 功能 | 适用场景 |
|------|------|----------|
| `add_user_fields.py` | 添加 username、role、permissions 字段 | v1.0 → v1.1+ |
| `add_encrypted_key.py` | 添加 encrypted_key 字段，支持查看 API Key | v1.1 → v1.2+ |

**常见问题**：

1. **迁移脚本执行失败**
   - 检查数据库连接配置
   - 查看错误日志
   - 必要时手动执行 SQL

2. **旧数据不支持新功能**
   - API Key 查看功能：旧 Key 无法查看明文，需重新创建
   - 用户名字段：旧用户可能为空，可通过管理后台补充

## 三、详细安装脚本

### 3.1 主安装脚本 (install.sh)

```bash
#!/bin/bash
#==============================================================================
# API服务全自动安装脚本
# 版本: 1.0.0
# 作者: API Service Team
# 兼容: Ubuntu 20.04+, CentOS 8+, Debian 11+
#==============================================================================

set -e  # 遇到错误立即退出

#==============================================================================
# 配置变量
#==============================================================================
VERSION="1.0.0"
INSTALL_DIR="/opt/api-service"
DATA_DIR="/data/api-service"
LOG_FILE="/var/log/api-service-install.log"
BACKUP_DIR="/backup/api-service"

# 颜色输出
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m' # No Color

# 默认配置
DOMAIN="api.local"
DB_PASSWORD=""
REDIS_PASSWORD=""
ADMIN_EMAIL="admin@example.com"
INSTALL_MODE="allinone"
SKIP_BACKUP=false
OFFLINE_MODE=false

#==============================================================================
# 函数定义
#==============================================================================

log() {
    local level=$1
    shift
    local message="$@"
    local timestamp=$(date '+%Y-%m-%d %H:%M:%S')
    echo -e "${timestamp} [${level}] ${message}" | tee -a "$LOG_FILE"
}

log_info() { log "INFO" "${BLUE}$1${NC}"; }
log_success() { log "SUCCESS" "${GREEN}$1${NC}"; }
log_warning() { log "WARNING" "${YELLOW}$1${NC}"; }
log_error() { log "ERROR" "${RED}$1${NC}"; exit 1; }

# 检查是否为root用户
check_root() {
    if [[ $EUID -ne 0 ]]; then
        log_error "请使用root用户或sudo执行此脚本"
    fi
}

# 检查系统兼容性
check_system() {
    log_info "检查系统兼容性..."
    
    if [[ -f /etc/os-release ]]; then
        . /etc/os-release
        OS=$ID
        VERSION_ID=$VERSION_ID
    else
        log_error "无法检测操作系统类型"
    fi
    
    case $OS in
        ubuntu|debian)
            PKG_MANAGER="apt-get"
            ;;
        centos|rhel|rocky|alinux)
            PKG_MANAGER="yum"
            ;;
        *)
            log_error "不支持的操作系统: $OS"
            ;;
    esac
    
    log_success "检测到操作系统: $OS $VERSION_ID, 包管理器: $PKG_MANAGER"
}

# 生成随机密码
generate_password() {
    local length=${1:-32}
    openssl rand -base64 $length | tr -d '/+' | cut -c1-$length
}

# 解析命令行参数
parse_arguments() {
    while [[ $# -gt 0 ]]; do
        case $1 in
            --domain)
                DOMAIN="$2"
                shift 2
                ;;
            --db-password)
                DB_PASSWORD="$2"
                shift 2
                ;;
            --redis-password)
                REDIS_PASSWORD="$2"
                shift 2
                ;;
            --admin-email)
                ADMIN_EMAIL="$2"
                shift 2
                ;;
            --install-dir)
                INSTALL_DIR="$2"
                shift 2
                ;;
            --mode)
                INSTALL_MODE="$2"
                shift 2
                ;;
            --offline)
                OFFLINE_MODE=true
                shift
                ;;
            --skip-backup)
                SKIP_BACKUP=true
                shift
                ;;
            --help|-h)
                show_help
                exit 0
                ;;
            *)
                log_error "未知参数: $1"
                ;;
        esac
    done
    
    # 生成默认密码
    [[ -z "$DB_PASSWORD" ]] && DB_PASSWORD=$(generate_password 24)
    [[ -z "$REDIS_PASSWORD" ]] && REDIS_PASSWORD=$(generate_password 16)
}

# 显示帮助信息
show_help() {
    cat << EOF
API服务全自动安装脚本 v${VERSION}

用法: 
    ./install.sh [选项]

选项:
    --domain string         API域名 (默认: api.local)
    --db-password string   PostgreSQL密码 (自动生成如不指定)
    --redis-password string Redis密码 (自动生成如不指定)
    --admin-email string   管理员邮箱 (默认: admin@example.com)
    --install-dir string   安装目录 (默认: /opt/api-service)
    --mode string           安装模式: allinone|distributed (默认: allinone)
    --offline              离线安装模式
    --skip-backup          跳过备份步骤
    --help, -h             显示此帮助信息

示例:
    ./install.sh                                    # 默认安装
    ./install.sh --domain api.example.com            # 自定义域名
    ./install.sh --mode distributed                  # 分布式安装
    ./install.sh --offline                          # 离线安装

EOF
}

# 备份原有数据
backup_existing() {
    if [[ "$SKIP_BACKUP" == true ]]; then
        log_info "跳过备份步骤"
        return
    fi
    
    log_info "检查原有数据并备份..."
    
    if [[ -d "$INSTALL_DIR" ]]; then
        log_warning "检测到原有安装，开始备份..."
        mkdir -p "$BACKUP_DIR"
        local backup_file="${BACKUP_DIR}/backup_$(date +%Y%m%d_%H%M%S).tar.gz"
        tar -czf "$backup_file" -C "$(dirname $INSTALL_DIR)" "$(basename $INSTALL_DIR)" 2>/dev/null || true
        log_success "备份已保存至: $backup_file"
    fi
}

# 安装系统依赖
install_dependencies() {
    log_info "安装系统依赖..."
    
    if [[ "$PKG_MANAGER" == "apt-get" ]]; then
        apt-get update
        apt-get install -y \
            curl \
            wget \
            git \
            vim \
            htop \
            net-tools \
            ca-certificates \
            gnupg \
            lsb-release \
            unzip
    else
        yum install -y \
            curl \
            wget \
            git \
            vim \
            htop \
            net-tools \
            ca-certificates \
            yum-utils \
            device-mapper-persistent-data \
            lvm2 \
            unzip
    fi
    
    log_success "系统依赖安装完成"
}

# 安装Docker
install_docker() {
    log_info "安装Docker..."
    
    if command -v docker &> /dev/null; then
        log_info "Docker已安装，版本: $(docker --version)"
        return
    fi
    
    if [[ "$PKG_MANAGER" == "apt-get" ]]; then
        # 添加Docker官方GPG密钥
        mkdir -p /etc/apt/keyrings
        curl -fsSL https://download.docker.com/linux/$OS/gpg | gpg --dearmor -o /etc/apt/keyrings/docker.gpg
        
        # 添加Docker仓库
        echo "deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.gpg] https://download.docker.com/linux/$OS $(lsb_release -cs) stable" | tee /etc/apt/sources.list.d/docker.list > /dev/null
        
        # 安装Docker
        apt-get update
        apt-get install -y docker-ce docker-ce-cli containerd.io docker-buildx-plugin docker-compose-plugin
    else
        # CentOS/RHEL
        yum-config-manager --add-repo https://download.docker.com/linux/centos/docker-ce.repo
        yum install -y docker-ce docker-ce-cli containerd.io docker-buildx-plugin docker-compose-plugin
    fi
    
    # 启动Docker
    systemctl start docker
    systemctl enable docker
    
    log_success "Docker安装完成，版本: $(docker --version)"
}

# 配置防火墙
configure_firewall() {
    log_info "配置防火墙..."
    
    # UFW防火墙
    if command -v ufw &> /dev/null; then
        ufw allow 22/tcp
        ufw allow 80/tcp
        ufw allow 443/tcp
        ufw --force enable
        log_info "UFW防火墙已配置"
    fi
    
    # firewalld防火墙
    if command -v firewall-cmd &> /dev/null; then
        firewall-cmd --permanent --add-port=22/tcp
        firewall-cmd --permanent --add-port=80/tcp
        firewall-cmd --permanent --add-port=443/tcp
        firewall-cmd --reload
        log_info "firewalld防火墙已配置"
    fi
    
    log_success "防火墙配置完成"
}

# 创建Docker Compose配置
create_docker_compose() {
    log_info "创建Docker Compose配置..."
    
    cat > "$INSTALL_DIR/docker-compose.yml" << 'EOF'
#==============================================================================
# API服务Docker Compose配置
# 版本: ${VERSION}
#==============================================================================

version: '3.8'

services:
  # PostgreSQL 数据库
  postgres:
    image: postgres:15-alpine
    container_name: api-postgres
    restart: unless-stopped
    environment:
      POSTGRES_DB: ${POSTGRES_DB}
      POSTGRES_USER: ${POSTGRES_USER}
      POSTGRES_PASSWORD: ${POSTGRES_PASSWORD}
      PGDATA: /var/lib/postgresql/data/pgdata
    volumes:
      - ${DATA_DIR}/postgres:/var/lib/postgresql/data
      - ${INSTALL_DIR}/config/init.sql:/docker-entrypoint-initdb.d/init.sql:ro
    ports:
      - "5432:5432"
    networks:
      - api-network
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U ${POSTGRES_USER}"]
      interval: 10s
      timeout: 5s
      retries: 5

  # Redis 缓存
  redis:
    image: redis:7-alpine
    container_name: api-redis
    restart: unless-stopped
    command: redis-server --requirepass ${REDIS_PASSWORD} --appendonly yes
    volumes:
      - ${DATA_DIR}/redis:/data
    ports:
      - "6379:6379"
    networks:
      - api-network
    healthcheck:
      test: ["CMD", "redis-cli", "-a", "${REDIS_PASSWORD}", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5

  # MinIO 对象存储
  minio:
    image: minio/minio:latest
    container_name: api-minio
    restart: unless-stopped
    environment:
      MINIO_ROOT_USER: ${MINIO_ROOT_USER}
      MINIO_ROOT_PASSWORD: ${MINIO_ROOT_PASSWORD}
    volumes:
      - ${DATA_DIR}/minio:/data
    ports:
      - "9000:9000"
      - "9001:9001"
    networks:
      - api-network
    command: server /data --console-address ":9001"
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:9000/minio/health/live"]
      interval: 30s
      timeout: 20s
      retries: 3

  # Nginx 反向代理
  nginx:
    image: nginx:alpine
    container_name: api-nginx
    restart: unless-stopped
    volumes:
      - ${INSTALL_DIR}/nginx/nginx.conf:/etc/nginx/nginx.conf:ro
      - ${INSTALL_DIR}/nginx/conf.d:/etc/nginx/conf.d:ro
      - ${INSTALL_DIR}/ssl:/etc/nginx/ssl:ro
      - ${DATA_DIR}/logs/nginx:/var/log/nginx
    ports:
      - "80:80"
      - "443:443"
    depends_on:
      - api
      - admin
    networks:
      - api-network
    healthcheck:
      test: ["CMD", "nginx", "-t"]
      interval: 30s
      timeout: 10s
      retries: 3

  # API服务
  api:
    build:
      context: .
      dockerfile: Dockerfile
    container_name: api-service
    restart: unless-stopped
    env_file:
      - ${INSTALL_DIR}/.env
    volumes:
      - ${DATA_DIR}/uploads:/app/uploads
      - ${DATA_DIR}/logs:/app/logs
    ports:
      - "8000:8000"
    depends_on:
      postgres:
        condition: service_healthy
      redis:
        condition: service_healthy
      minio:
        condition: service_healthy
    networks:
      - api-network
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8000/health"]
      interval: 30s
      timeout: 10s
      retries: 3

  # 管理后台
  admin:
    image: nginx:alpine
    container_name: api-admin
    restart: unless-stopped
    volumes:
      - ${INSTALL_DIR}/admin:/usr/share/nginx/html:ro
      - ${DATA_DIR}/logs/admin:/var/log/nginx
    depends_on:
      - api
    networks:
      - api-network
    healthcheck:
      test: ["CMD", "wget", "--quiet", "--tries=1", "--spider", "http://localhost/"]
      interval: 30s
      timeout: 10s
      retries: 3

networks:
  api-network:
    driver: bridge

volumes:
  postgres_data:
  redis_data:
  minio_data:
EOF
    
    log_success "Docker Compose配置创建完成"
}

# 创建Nginx配置
create_nginx_config() {
    log_info "创建Nginx配置..."
    
    cat > "$INSTALL_DIR/nginx/nginx.conf" << 'EOF'
user nginx;
worker_processes auto;
error_log /var/log/nginx/error.log warn;
pid /var/run/nginx.pid;

events {
    worker_connections 1024;
    use epoll;
    multi_accept on;
}

http {
    include /etc/nginx/mime.types;
    default_type application/octet-stream;

    log_format main '$remote_addr - $remote_user [$time_local] "$request" '
                    '$status $body_bytes_sent "$http_referer" '
                    '"$http_user_agent" "$http_x_forwarded_for"';

    access_log /var/log/nginx/access.log main;

    sendfile on;
    tcp_nopush on;
    tcp_nodelay on;
    keepalive_timeout 65;
    types_hash_max_size 2048;
    client_max_body_size 100M;

    # Gzip压缩
    gzip on;
    gzip_vary on;
    gzip_proxied any;
    gzip_comp_level 6;
    gzip_types text/plain text/css text/xml text/javascript 
               application/json application/javascript application/xml+rss;

    include /etc/nginx/conf.d/*.conf;
}
EOF

    cat > "$INSTALL_DIR/nginx/conf.d/api.conf" << EOF
#==============================================================================
# API服务Nginx配置
# 域名: ${DOMAIN}
#==============================================================================

upstream api_backend {
    server api:8000;
    keepalive 32;
}

upstream admin_backend {
    server admin:80;
    keepalive 16;
}

# HTTP重定向到HTTPS
server {
    listen 80;
    server_name ${DOMAIN};
    
    # Let's Encrypt证书验证
    location /.well-known/acme-challenge/ {
        root /var/www/certbot;
    }
    
    location / {
        return 301 https://\$host\$request_uri;
    }
}

# HTTPS主站
server {
    listen 443 ssl http2;
    server_name ${DOMAIN};

    # SSL证书 (Let's Encrypt)
    ssl_certificate /etc/nginx/ssl/fullchain.pem;
    ssl_certificate_key /etc/nginx/ssl/privkey.pem;
    
    # SSL安全配置
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256;
    ssl_prefer_server_ciphers off;
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 1d;

    # 安全响应头
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;

    # API文档
    location /docs {
        proxy_pass http://api_backend/docs;
        proxy_set_header Host \$host;
        proxy_set_header X-Real-IP \$remote_addr;
        proxy_set_header X-Forwarded-For \$proxy_add_x_forwarded_for;
    }

    # Swagger UI
    location /swagger {
        proxy_pass http://api_backend/swagger-ui.html;
        proxy_set_header Host \$host;
        proxy_set_header X-Real-IP \$remote_addr;
        proxy_http_version 1.1;
        proxy_set_header Connection "";
    }

    # API接口
    location /api/ {
        # 限流
        limit_req zone=api_limit burst=20 nodelay;
        
        proxy_pass http://api_backend/api/;
        proxy_http_version 1.1;
        proxy_set_header Host \$host;
        proxy_set_header X-Real-IP \$remote_addr;
        proxy_set_header X-Forwarded-For \$proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto \$scheme;
        proxy_set_header Connection "";
        
        # 超时配置
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;
    }

    # 管理后台
    location /admin/ {
        proxy_pass http://admin_backend/;
        proxy_http_version 1.1;
        proxy_set_header Host \$host;
        proxy_set_header X-Real-IP \$remote_addr;
        proxy_set_header X-Forwarded-For \$proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto \$scheme;
        proxy_set_header Connection "";
    }

    # 健康检查
    location /health {
        access_log off;
        return 200 "healthy\n";
        add_header Content-Type text/plain;
    }

    # 静态资源缓存
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2|ttf|eot)$ {
        expires 30d;
        add_header Cache-Control "public, immutable";
    }

    # 日志
    access_log /var/log/nginx/api_access.log main;
    error_log /var/log/nginx/api_error.log;
}
EOF
    
    log_success "Nginx配置创建完成"
}

# 创建数据库初始化脚本
create_database_init() {
    log_info "创建数据库初始化脚本..."
    
    cat > "$INSTALL_DIR/config/init.sql" << 'EOF'
--==============================================================================
-- API服务数据库初始化脚本
-- 版本: 1.0.0
--==============================================================================

-- 启用扩展
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";
CREATE EXTENSION IF NOT EXISTS "pg_trgm";

-- 创建表
CREATE TABLE IF NOT EXISTS tenants (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    name VARCHAR(100) NOT NULL UNIQUE,
    domain VARCHAR(255) UNIQUE,
    status VARCHAR(20) DEFAULT 'active',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS api_keys (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    tenant_id UUID REFERENCES tenants(id) ON DELETE CASCADE,
    key_prefix VARCHAR(8) NOT NULL,
    key_hash VARCHAR(64) NOT NULL UNIQUE,
    name VARCHAR(100) NOT NULL,
    description TEXT,
    rate_limit INTEGER DEFAULT 100,
    status VARCHAR(20) DEFAULT 'active',
    expires_at TIMESTAMP,
    last_used_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS call_logs (
    id BIGSERIAL PRIMARY KEY,
    api_key_id UUID REFERENCES api_keys(id) ON DELETE SET NULL,
    method VARCHAR(10) NOT NULL,
    endpoint VARCHAR(255) NOT NULL,
    status_code INTEGER,
    response_time INTEGER,
    ip_address INET,
    user_agent TEXT,
    request_size INTEGER,
    response_size INTEGER,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS rate_limits (
    id BIGSERIAL PRIMARY KEY,
    api_key_id UUID REFERENCES api_keys(id) ON DELETE CASCADE,
    window_start TIMESTAMP NOT NULL,
    request_count INTEGER DEFAULT 0,
    UNIQUE(api_key_id, window_start)
);

CREATE INDEX idx_call_logs_created_at ON call_logs(created_at);
CREATE INDEX idx_call_logs_api_key_id ON call_logs(api_key_id);
CREATE INDEX idx_api_keys_tenant_id ON api_keys(tenant_id);
CREATE INDEX idx_rate_limits_api_key_id ON rate_limits(api_key_id);

-- 插入默认租户
INSERT INTO tenants (name, domain, status) 
VALUES ('default', 'default.local', 'active')
ON CONFLICT (name) DO NOTHING;
EOF
    
    log_success "数据库初始化脚本创建完成"
}

# 拉取和启动服务

# 配置SSL证书
configure_ssl() {
    log_info "配置SSL证书..."
    
    # 创建SSL目录
    mkdir -p "$INSTALL_DIR/ssl"
    
    # 使用自签名证书（生产环境请使用Let's Encrypt）
    if [[ ! -f "$INSTALL_DIR/ssl/privkey.pem" ]]; then
        log_warning "使用自签名证书，生产环境请配置正式证书"
        openssl req -x509 -nodes -days 365 -newkey rsa:2048 \
            -keyout "$INSTALL_DIR/ssl/privkey.pem" \
            -out "$INSTALL_DIR/ssl/fullchain.pem" \
            -subj "/C=CN/ST=Beijing/L=Beijing/O=API Service/CN=${DOMAIN}"
    fi
    
    log_success "SSL证书配置完成"
}

# 创建验证脚本
create_verify_script() {
    log_info "创建服务验证脚本..."
    
    cat > "$INSTALL_DIR/verify-deployment.sh" << 'EOF'
#!/bin/bash
#==============================================================================
# API服务部署验证脚本
#==============================================================================

RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'

check_service() {
    local name=$1
    local url=$2
    
    echo -n "检查 $name... "
    if curl -sf "$url" > /dev/null 2>&1; then
        echo -e "${GREEN}✓ 通过${NC}"
        return 0
    else
        echo -e "${RED}✗ 失败${NC}"
        return 1
    fi
}

echo "========================================"
echo "API服务部署验证"
echo "========================================"
echo ""

failed=0

# 检查Docker容器
echo "1. Docker容器状态:"
docker ps --format "table {{.Names}}\t{{.Status}}" | grep -E "api|postgres|redis|minio|nginx" || true
echo ""

# 检查服务端口
echo "2. 服务端口检测:"
netstat -tlnp 2>/dev/null | grep -E ":(80|443|5432|6379|8000|9000)" || ss -tlnp | grep -E ":(80|443|5432|6379|8000|9000)" || true
echo ""

# 检查服务健康
check_service "Nginx" "http://localhost/health" || ((failed++))
check_service "API服务" "http://localhost:8000/health" || ((failed++))
check_service "管理后台" "http://localhost/admin/" || ((failed++))
check_service "API文档" "http://localhost/docs/" || ((failed++))
echo ""

# 检查数据库连接
echo "3. 数据库连接检测:"
if docker exec api-postgres pg_isready -U api_user > /dev/null 2>&1; then
    echo -e "   PostgreSQL ${GREEN}✓ 连接正常${NC}"
else
    echo -e "   PostgreSQL ${RED}✗ 连接失败${NC}"
    ((failed++))
fi

# 检查Redis连接
echo "4. Redis连接检测:"
if docker exec api-redis redis-cli -a ${REDIS_PASSWORD:-''} ping > /dev/null 2>&1; then
    echo -e "   Redis ${GREEN}✓ 连接正常${NC}"
else
    echo -e "   Redis ${RED}✗ 连接失败${NC}"
    ((failed++))
fi

echo ""
echo "========================================"
if [[ $failed -eq 0 ]]; then
    echo -e "${GREEN}所有检查通过！部署成功！${NC}"
    echo ""
    echo "访问地址:"
    echo "  API地址: https://localhost/api/"
    echo "  管理后台: https://localhost/admin/"
    echo "  API文档: https://localhost/docs/"
    echo "  Swagger: https://localhost/swagger/"
    echo ""
    exit 0
else
    echo -e "${RED}有 $failed 项检查失败，请查看日志${NC}"
    exit 1
fi
EOF
    
    chmod +x "$INSTALL_DIR/verify-deployment.sh"
    log_success "验证脚本创建完成"
}
